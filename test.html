PROCEDURE PR_Data_in_acc_tab (cust_no varchar2) IS BEGIN dbms_output.put_line(cust_no); FOR rec in( select CA.BRANCH_CODE, CUST_NO, CA.CUST_AC_NO, CBR_AC_NO, RECORD_STAT from STTM_CUST_ACCOUNT CA JOIN STTM_CUST_ACCOUNT_CUSTOM CAC ON CA.CUST_AC_NO=CAC.CUST_AC_NO WHERE CUST_NO=cust_no --AND RECORD_STAT='O' ) LOOP dbms_output.put_line(cust_no); dbms_output.put_line(rec.CUST_AC_NO); dbms_output.put_line(rec.branch_code); dbms_output.put_line(rec.CBR_AC_NO); INSERT INTO STTB_STDCIFAC_ACCOUNT_INFO VALUES( rec.CUST_AC_NO, rec.branch_code, rec.CBR_AC_NO, '', '', '','' ); END LOOP; END PR_Data_in_acc_tab;
